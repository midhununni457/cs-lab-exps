-- packages.sql
-- Examples: CREATE PACKAGE and CREATE PACKAGE BODY

-- Package specification
CREATE OR REPLACE PACKAGE pkg_math AS
  FUNCTION add_numbers(a IN NUMBER, b IN NUMBER) RETURN NUMBER;
  PROCEDURE print_hello;
END pkg_math;
/
-- Package body
CREATE OR REPLACE PACKAGE BODY pkg_math AS
  FUNCTION add_numbers(a IN NUMBER, b IN NUMBER) RETURN NUMBER IS
  BEGIN
    RETURN a + b;
  END add_numbers;

  PROCEDURE print_hello IS
  BEGIN
    DBMS_OUTPUT.PUT_LINE('Hello from package');
  END print_hello;
END pkg_math;
/
-- Usage
DECLARE
  v NUMBER;
BEGIN
  v := pkg_math.add_numbers(2,3);
  DBMS_OUTPUT.PUT_LINE('v = ' || v);
  pkg_math.print_hello;
END;
/
